Python matplotlib example: versiones

# -*- coding: utf-8 -*-

import scipy as sp
import pandas as pd
import numpy as num

##Ver version instalada
print("Scipy:",sp.version.version)
print("Scipy:",sp.__version__)
print("NumPy:",num.__version__)
